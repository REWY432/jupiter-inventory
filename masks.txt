        if (!isAutoSerialEnabled) return; // Don't auto-update if disabled (e.g. editing old record)

        const model = document.getElementById('model').value;

        const month = String(document.getElementById('month').value).padStart(2, '0');
        const yearFull = document.getElementById('year').value;

    window.handleShipmentSearch = function(query) {
        const resultsDiv = document.getElementById('shipmentSearchResults');
        if (!query || query.trim() === '') {
            resultsDiv.classList.add('d-none');
            return;
        }
        query = query.toLowerCase();
        
        // Filter DB: not in buffer AND matches query
        const matches = window.localDB.filter(item => {
            const isInBuffer = window.shipmentBuffer.some(b => b.id === item.id);
            const matchesQuery = item.serial.toLowerCase().includes(query) || (item.globalSeq && String(item.globalSeq).includes(query));
            return !isInBuffer && matchesQuery;
        }).slice(0, 10); // Limit to top 10 matches

        if (matches.length === 0) {
            resultsDiv.classList.remove('d-none');
            resultsDiv.innerHTML = '<div class="p-3 text-muted small text-center">Ничего не найдено</div>';
            return;
        }

        let html = '';
        matches.forEach(item => {
            html += `
            <div class="shipment-item" style="cursor:pointer;" onclick="addToShipment('${item.id}')">
                <div>
                    <span class="fw-bold text-primary">${item.serial}</span>
                    <span class="text-muted small ms-2">${item.model}, ${item.size}</span>
                </div>
                <i class="fas fa-plus text-success"></i>
            </div>`;
        });
        resultsDiv.innerHTML = html;
        resultsDiv.classList.remove('d-none');
    }

    function getPageData() {
        const search = (document.getElementById('searchSerial').value || "").toLowerCase();
        const fModel = document.getElementById('filterModel').value;
        const fSize = document.getElementById('filterSize').value;
        const fCRM = document.getElementById('filterCRM').value;
        
        let filtered = window.localDB.filter(i => {
            const matchesSearch = (i.serial.toLowerCase().includes(search) || (i.notes||"").toLowerCase().includes(search));
            const matchesModel = fModel === "" || i.model === fModel;
            const matchesSize = fSize === "" || i.size === fSize;
            let matchesCRM = true;
            if (fCRM === '1') matchesCRM = !!i.isInCRM;
            if (fCRM === '0') matchesCRM = !i.isInCRM;
            return matchesSearch && matchesModel && matchesSize && matchesCRM;
        });

        filtered.sort((a, b) => {
            let valA, valB;
            if (currentSort.field === 'serial') { valA = a.serial.toLowerCase(); valB = b.serial.toLowerCase(); } 
            else if (currentSort.field === 'model') { valA = a.model.toLowerCase(); valB = b.model.toLowerCase(); } 
            else if (currentSort.field === 'date') { valA = (a.year * 100) + a.month; valB = (b.year * 100) + b.month; } 
            else { valA = a.globalSeq || 0; valB = b.globalSeq || 0; }
            if (valA < valB) return currentSort.direction === 'asc' ? -1 : 1;
            if (valA > valB) return currentSort.direction === 'asc' ? 1 : -1;
            return 0;
        });

        const totalPages = Math.ceil(filtered.length / itemsPerPage) || 1;
        if(currentPage > totalPages) currentPage = 1;
        const start = (currentPage - 1) * itemsPerPage;
        return filtered.slice(start, start + itemsPerPage);
    }

    function addSwipeListeners() {
        if(window.innerWidth > 991) return; // Only mobile
        const rows = document.querySelectorAll('.table tr');
        
        rows.forEach(row => {
            let startX, currentX;
            const content = row.querySelectorAll('.mobile-row-content'); // We move all content cells
            
            row.addEventListener('touchstart', (e) => {
                startX = e.touches[0].clientX;
            }, {passive: true});
            
            row.addEventListener('touchmove', (e) => {
                currentX = e.touches[0].clientX;
                const diff = currentX - startX;
                // Only allow swiping left (negative diff), max -140px
                if(diff < 0 && diff > -160) {
                    content.forEach(el => el.style.transform = `translateX(${diff}px)`);
                }
            }, {passive: true});
            
            row.addEventListener('touchend', (e) => {
                const diff = currentX - startX;
                if(diff < -70) { // Threshold to open
                    content.forEach(el => el.style.transform = `translateX(-140px)`);
                } else {
                    content.forEach(el => el.style.transform = `translateX(0)`);
                }
            });
        });
    }

    window.importData = function(input) {
        const file = input.files[0];
        if(!file) return;
        
        Papa.parse(file, {
            header: true,
            skipEmptyLines: true,
            complete: function(results) {
                if(results.data && results.data.length > 0) {
                    processImportData(results.data);
                } else {
                    showToast('Файл пуст или не распознан', 'error');
                }
            },
            error: function(err) {
                showToast('Ошибка чтения CSV: ' + err.message, 'error');
            }
        });
        input.value = '';
    }

    function processImportData(data) {
        window.pendingImportData = [];
        let maxSeq = window.localDB.length > 0 ? Math.max(...window.localDB.map(o => o.globalSeq || 0)) : 0;
        
        // Flexible Mapping
        data.forEach(row => {
            // Try different case keys
            const getVal = (keys) => {
                for(let k of keys) {
                    const found = Object.keys(row).find(rk => rk.toLowerCase().trim() === k.toLowerCase());
                    if(found) return row[found];
                }
                return null;
            };

            const serial = getVal(['Serial', 'Серийный', 'SN', 'Mask']) || `AUTO-IMP-${Math.random().toString(36).substr(2,5)}`;
            const model = getVal(['Model', 'Модель']) || "Рапира";
            const size = getVal(['Size', 'Размер']) || "M";
            
            // Validate existing
            const exists = window.localDB.some(x => x.serial === serial);
            if(exists) return; // Skip duplicates
            
            maxSeq++;
            window.pendingImportData.push({
                serial: serial,
                globalSeq: maxSeq,
                model: model,
                size: size,
                generation: getVal(['Gen', 'Generation', 'Поколение']) || "-",
                lining: getVal(['Lining', 'Подкладка']) || "-",
                year: Number(getVal(['Year', 'Год'])) || new Date().getFullYear(),
                month: Number(getVal(['Month', 'Месяц'])) || new Date().getMonth()+1,
                notes: "Imported via CSV",
                isInCRM: false
            });
        });
        
        // Render Preview
        const tbody = document.getElementById('importPreviewBody');
        if(tbody) {
            tbody.innerHTML = '';
            window.pendingImportData.slice(0, 50).forEach(d => {
                tbody.innerHTML += `<tr><td>${d.serial}</td><td>${d.model}</td><td>${d.size}</td><td>${d.generation}</td><td>${d.month}.${d.year}</td></tr>`;
            });
        }
        
        const stats = document.getElementById('importStats');
        if(stats) stats.innerText = `Готово к импорту: ${window.pendingImportData.length} (Дубликаты пропущены)`;
        
        new bootstrap.Modal(document.getElementById('importModal')).show();
    }

    window.executeDelete = async function() {
        // If modal is open, hide it
        const modalEl = document.getElementById('deleteModal');
        const modal = bootstrap.Modal.getInstance(modalEl);
        if(modal) modal.hide();

        // 1. Visually hide rows immediately
        if(isBulkDelete) {
            selectedIds.forEach(id => {
                const el = document.getElementById(`row-${id}`);
                if(el) el.style.display = 'none';
            });
        } else {
            const el = document.getElementById(`row-${recordToDeleteId}`);
            if(el) el.style.display = 'none';
        }

        // 2. Show Toast with Undo
        const toastEl = document.createElement('div');
        toastEl.className = `toast show align-items-center bg-dark text-white border-0 shadow`;
        toastEl.innerHTML = `
            <div class="d-flex w-100 justify-content-between p-3 align-items-center">
                <span>Удалено ${isBulkDelete ? selectedIds.size : '1'} зап.</span>
                <button class="btn btn-sm btn-warning fw-bold" onclick="undoDelete()">ОТМЕНА</button>
            </div>`;
        document.getElementById('toastContainer').appendChild(toastEl);
        
        // 3. Set Timer for actual delete
        undoTimer = setTimeout(async () => {
            try {
                if(isBulkDelete) {
                    const promises = Array.from(selectedIds).map(id => deleteDoc(doc(dbRef, "masks", id)));
                    await Promise.all(promises);
                    clearSelection();
                } else {
                    await deleteDoc(doc(dbRef, "masks", recordToDeleteId));
                }
                toastEl.remove();
            } catch (e) {
                console.error(e);
            }
        }, 5000); // 5 seconds to undo
    }

    window.undoDelete = function() {

        if(isBulkDelete) {
            selectedIds.forEach(id => {
                const el = document.getElementById(`row-${id}`);
                if(el) el.style.display = '';
            });
        } else {
            const el = document.getElementById(`row-${recordToDeleteId}`);
